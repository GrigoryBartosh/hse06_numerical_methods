def max_eig(A, p=100):
    b = np.ones(A.shape[0])
    for i in range(p):
        b = A @ b
        b /= np.linalg.norm(b)
    return np.inner(A @ b, b) / np.inner(b, b)

def max_eig_eitken(A, p=100):
    b = np.ones(A.shape[0])
    s1 = np.inner(A @ b, b) / np.inner(b, b)
    b = A @ b
    s2 = np.inner(A @ b, b) / np.inner(b, b)
    b = A @ b
    s3 = np.inner(A @ b, b) / np.inner(b, b)
    mx = s3 - (s3 - s2) ** 2 / (s3 - s2 - s2 + s1)
    for i in range(3, p):
        b = A @ b
        s1, s2, s3 = s2, s3, np.inner(A @ b, b) / np.inner(b, b)
        mx = s3 - (s3 - s2) ** 2 / (s3 - 2 * s2 + s1)
    return mx

def min_eig_eitken(A, p=100, norm=False):
    t = max_eig_eitken(A, p=p)
    return max_eig_eitken(A - t * np.identity(A.shape[0]), p=p) + t
